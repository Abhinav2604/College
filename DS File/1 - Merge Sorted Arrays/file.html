int merge(int a[] , int b[], int c[] , int lena , int lenb){

    int c1 = 0 ;
    int c2 = 0 ;
    int i = 0 ;
    while(c1 < lena && c2 < lenb){
        if(a[c1] > b[c2]) {
            c[i++] = b[c2++] ;
        } else {
            c[i++] = a[c1++] ;
        }
    }
    while(i < lena  + lenb ) {
        if(c1<lena) {
            c[i++] = a[c1++] ;
        } else {
            c[i++] = b[c2++] ;
        }
    }
}
